## Minimal REGIONAL

The simplest possible configuration. Attaches the six default AWS managed rule groups and takes no action on non-matching requests (default action: `allow`).

```hcl
module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name  = "my-web-acl"
  scope = "REGIONAL"
}
```

## REGIONAL with ALB association

Creates the Web ACL and associates it with an existing Application Load Balancer.

```hcl
module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name          = "alb-web-acl"
  scope         = "REGIONAL"
  associate_alb = true
  alb_arn       = aws_lb.main.arn
}
```

## CloudFront

CloudFront Web ACLs must be created in `us-east-1`. Pass an aliased provider to satisfy this requirement.

```hcl
provider "aws" {
  alias  = "us-east-1"
  region = "us-east-1"
}

module "cloudfront_wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"
  providers = {
    aws = aws.us-east-1
  }

  name  = "cloudfront-web-acl"
  scope = "CLOUDFRONT"
}
```

## IP blocking

Create an IP set and reference it from the module to block specific addresses.

```hcl
resource "aws_wafv2_ip_set" "blocked" {
  name               = "blocked-ips"
  scope              = "REGIONAL"
  ip_address_version = "IPV4"
  addresses          = ["1.2.3.4/32", "5.6.7.8/32"]
}

module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name  = "my-web-acl"
  scope = "REGIONAL"

  ip_sets_rule = [
    {
      name       = "blocked-ips"
      action     = "block"
      priority   = 1
      ip_set_arn = aws_wafv2_ip_set.blocked.arn
    }
  ]
}
```

## Rate limiting

Apply a global per-IP rate limit and an additional tighter limit scoped to a specific URL prefix.

```hcl
module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name  = "my-web-acl"
  scope = "REGIONAL"

  ip_rate_based_rule = {
    name     = "ip-rate-limit"
    priority = 7
    action   = "block"
    limit    = 100
  }

  ip_rate_url_based_rules = [
    {
      name                  = "api-rate-limit"
      priority              = 8
      action                = "block"
      limit                 = 50
      search_string         = "/api/"
      positional_constraint = "STARTS_WITH"
    }
  ]
}
```

## Logging to CloudWatch

Deliver WAF logs to a CloudWatch Logs log group. The log group name must begin with `aws-waf-logs-`.

```hcl
resource "aws_cloudwatch_log_group" "waf_logs" {
  name              = "aws-waf-logs-my-web-acl"
  retention_in_days = 30
}

module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name  = "my-web-acl"
  scope = "REGIONAL"

  enable_logging = true
  log_destination_arns = [
    aws_cloudwatch_log_group.waf_logs.arn
  ]
}
```

## Complete ALB example

The following example is drawn from the module's own test fixtures. It demonstrates all rule types used together: managed rules with a rule-level action override, header filtering, IP set rules, a global rate-based rule, URL-scoped rate-based rules, and a custom rule group.

```hcl
resource "aws_wafv2_ip_set" "block_all_ips" {
  name               = "block-all-ips"
  scope              = "REGIONAL"
  ip_address_version = "IPV4"
  addresses          = formatlist("%s.0.0.0/8", range(0, 256))
}

resource "aws_wafv2_ip_set" "ipset" {
  name               = "my-ip-set"
  scope              = "REGIONAL"
  ip_address_version = "IPV4"
  addresses = [
    "1.2.3.4/32",
    "5.6.7.8/32"
  ]
}

resource "aws_wafv2_rule_group" "block_countries" {
  name     = "block-countries"
  scope    = "REGIONAL"
  capacity = 1

  rule {
    name     = "rule-1"
    priority = 1

    action {
      block {}
    }

    statement {
      geo_match_statement {
        country_codes = ["UA"]
      }
    }

    visibility_config {
      cloudwatch_metrics_enabled = false
      metric_name                = "friendly-rule-metric-name"
      sampled_requests_enabled   = false
    }
  }

  visibility_config {
    cloudwatch_metrics_enabled = false
    metric_name                = "friendly-metric-name"
    sampled_requests_enabled   = false
  }
}

module "wafv2" {
  source  = "trussworks/wafv2/aws"
  version = "~> 3.0"

  name          = "my-web-acl"
  scope         = "REGIONAL"
  associate_alb = true
  alb_arn       = aws_lb.main.arn

  managed_rules = [
    {
      name            = "AWSManagedRulesAmazonIpReputationList"
      override_action = "none"
      priority        = 1
      vendor_name     = "AWS"
      rule_action_override = []
    },
    {
      name            = "AWSManagedRulesCommonRuleSet"
      override_action = "none"
      priority        = 2
      vendor_name     = "AWS"
      rule_action_override = [
        {
          name          = "SizeRestrictions_BODY"
          action_to_use = "allow"
        }
      ]
    },
    {
      name            = "AWSManagedRulesSQLiRuleSet"
      override_action = "none"
      priority        = 3
      vendor_name     = "AWS"
      rule_action_override = []
    }
  ]

  filtered_header_rule = {
    header_types  = ["test1", "test2"]
    header_value  = "host"
    priority      = 1
    action        = "allow"
    search_string = ""
  }

  ip_sets_rule = [
    {
      name       = "my-ip-set"
      priority   = 5
      action     = "count"
      ip_set_arn = aws_wafv2_ip_set.ipset.arn
    },
    {
      name       = "block-all-ips"
      priority   = 6
      action     = "block"
      ip_set_arn = aws_wafv2_ip_set.block_all_ips.arn
    }
  ]

  ip_rate_based_rule = {
    name     = "ip-rate-limit"
    priority = 7
    action   = "block"
    limit    = 100
  }

  ip_rate_url_based_rules = [
    {
      name                  = "ip-rate-foo-limit"
      priority              = 8
      action                = "block"
      limit                 = 100
      search_string         = "/foo/"
      positional_constraint = "STARTS_WITH"
    },
    {
      name                  = "ip-rate-bar-limit"
      priority              = 9
      action                = "block"
      limit                 = 200
      search_string         = "/bar/"
      positional_constraint = "STARTS_WITH"
    }
  ]

  group_rules = [
    {
      name            = aws_wafv2_rule_group.block_countries.name
      arn             = aws_wafv2_rule_group.block_countries.arn
      override_action = "none"
      priority        = 11
    }
  ]
}
```
